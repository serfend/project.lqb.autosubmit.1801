bool isprime(int num)
{
	for (int i = 2; i <= sqrt(num); ++i)
	{
		if(num%i==0)
		{
			return false;
		}
	}
	return true;
}

void filter_prime()
{
	for (int i = 2; i <= n; ++i)
	{
		if(nums[i]==-1)		
		{
			continue;
		}
		else if (isprime(i))
		{
			nums[i]=1;
			cout<<i;
		}
		else
		{
			nums[i]=-1;
		} 

		for (int j = 2; i*j <= n; ++j)
		{
			nums[j*i]=-1;
		}
	}
}
