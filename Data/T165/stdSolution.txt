
#include <stdio.h>
main()
{
	double a;
	scanf("%lf",&a);
	if(a<=100000)
	{
		printf("%.2lf",a*0.1);
	}
	if(a<=200000&&a>100000)
	{
		printf("%.2lf",100000*0.1+(a-100000)*0.075);
	}
	if(a>200000&&a<=400000)
	{
		printf("%.2lf",100000*0.1+(100000*0.075)+(a-200000)*0.05);
	}
	if(a>400000&&a<=600000)
	{
		printf("%.2lf",100000*0.1+(100000*0.075)+(200000*0.05)+(a-400000)*0.03);
	}
	if(a>600000&&a<=1000000)
	{
		printf("%.2lf",100000*0.1+(100000*0.075)+(200000*0.05)+(200000*0.03)+(a-600000)*0.015);
	}
	if(a>1000000)
	{
		printf("%.2lf",100000*0.1+(100000*0.075)+(200000*0.05)+(200000*0.03)+(400000*0.015)+(a-1000000)*0.01);
	}
}
